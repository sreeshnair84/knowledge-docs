## 6. OWASP LLM Top 10 (2025) — SOC Controls

| # | Risk | Description | SOC-Specific Risk | Control Implementation |
|---|------|-------------|-------------------|----------------------|
| LLM01 | **Prompt Injection** | Malicious input overrides AI instructions | Attacker poisons alert data to manipulate AI triage | Data boundary enforcement; instruction hierarchy; output anomaly detection |
| LLM02 | **Sensitive Information Disclosure** | AI leaks training data or system prompts | SOC AI leaks detection playbooks, tool credentials | No credentials in prompts; system prompt confidentiality; output filtering |
| LLM03 | **Supply Chain** | Vulnerabilities in AI tools/libraries | Compromised Hugging Face model, poisoned pip package | Dependency scanning; model SBOM; private model registry |
| LLM04 | **Data and Model Poisoning** | Training data manipulation | Poisoned SOC incident data causes model misclassification | Training data provenance; behavioral baseline monitoring |
| LLM05 | **Improper Output Handling** | Unsafe AI output consumed by downstream systems | AI generates malicious KQL/SOAR code that gets auto-executed | Output sanitization; no auto-execution of AI-generated code |
| LLM06 | **Excessive Agency** | AI takes unauthorized high-impact actions | AI isolates 50 servers based on prompt injection | Least-privilege tools; rate limits; human approval gates |
| LLM07 | **System Prompt Leakage** | AI reveals confidential system prompt | Adversary learns SOC detection logic and evades it | System prompt confidentiality; monitor for extraction attempts |
| LLM08 | **Vector and Embedding Weaknesses** | Manipulated embeddings in RAG system | Poisoned playbook embeddings cause wrong playbook retrieval | Embedding provenance; signed embedding stores |
| LLM09 | **Misinformation** | AI generates false security intelligence | AI fabricates threat actor attribution | Hallucination detection; human review for attribution claims |
| LLM10 | **Unbounded Consumption** | AI resource abuse (denial of service) | Attacker floods SOC AI with expensive prompts | Rate limiting; token budgets per session; cost alerting |

```python
# OWASP LLM Top 10 control implementations

class OWASPLLMControlFramework:
    
    def llm01_prompt_injection_control(self, user_input: str, system_data: dict) -> dict:
        """LLM01: Prevent prompt injection"""
        # 1. Sanitize input
        sanitized = self.sanitize_injection_patterns(user_input)
        # 2. Use structured prompts with clear data boundaries
        prompt = self.build_boundary_enforced_prompt(sanitized, system_data)
        # 3. Monitor output for anomalies
        return {"prompt": prompt, "injection_risk_score": self.score_injection_risk(user_input)}
    
    def llm06_excessive_agency_control(self, requested_tool: str, params: dict) -> dict:
        """LLM06: Prevent excessive agency"""
        tool_config = TOOL_RISK_LEVELS.get(requested_tool, {"risk": "unknown"})
        
        if tool_config['risk'] == 'high' or tool_config.get('requires_approval'):
            return {
                "authorized": False,
                "reason": "Human approval required",
                "approval_request": self.create_approval_request(requested_tool, params)
            }
        
        if not self.check_rate_limit(requested_tool):
            return {"authorized": False, "reason": "Rate limit exceeded"}
        
        return {"authorized": True}
    
    def llm10_token_budget_control(self, agent_id: str, estimated_tokens: int) -> dict:
        """LLM10: Prevent unbounded consumption"""
        daily_budget = AGENT_TOKEN_BUDGETS.get(agent_id, 100000)
        daily_used = self.get_daily_token_usage(agent_id)
        
        if daily_used + estimated_tokens > daily_budget:
            return {
                "allowed": False,
                "reason": f"Daily token budget exceeded. Used: {daily_used}, Budget: {daily_budget}",
                "action": "queue_for_human_analyst"
            }
        
        return {"allowed": True, "remaining_budget": daily_budget - daily_used}
```

        "requirement": "Human oversight capability",
        "implementation": "Kill switch implementation; mandatory human approval for high-risk actions; override capability",
        "evidence": ["Human_Oversight_Architecture.pdf", "Kill_Switch_Test_Record.pdf"]
    },
    "Article_15_Accuracy_Robustness": {
        "requirement": "Accuracy, robustness, cybersecurity",
        "implementation": "Daily automated evaluation pipeline; quarterly red team; adversarial testing",
        "evidence": ["Accuracy_Evaluation_Reports/", "Red_Team_Report_Q1_2026.pdf"]
    }
}

def generate_eu_ai_act_conformity_assessment(soc_ai_system: dict) -> dict:
    """Generate conformity assessment checklist for EU AI Act high-risk systems"""
    
    checklist = {}
    for article, requirements in EU_AI_ACT_HIGH_RISK_REQUIREMENTS.items():
        evidence_complete = all(
            os.path.exists(f"compliance/{ev}") 
            for ev in requirements.get('evidence', [])
        )
        checklist[article] = {
            "requirement": requirements['requirement'],
            "implementation_documented": bool(requirements['implementation']),
            "evidence_complete": evidence_complete,
            "compliant": evidence_complete,
            "gap": None if evidence_complete else f"Missing evidence: {requirements['evidence']}"
        }
    
    compliant_count = sum(1 for v in checklist.values() if v['compliant'])
    
    return {
        "system_name": soc_ai_system['name'],
        "assessment_date": datetime.utcnow().date().isoformat(),
        "risk_classification": soc_ai_system.get('risk_classification', 'HIGH'),
        "article_compliance": checklist,
        "overall_compliance_rate": compliant_count / len(checklist),
        "ready_for_market": compliant_count == len(checklist)
    }

### Clause Mapping to AI SOC Operations

|-----------------|-------------|----------------------|

| **10. Improvement** | Nonconformity and improvement | Incident-driven improvement process; regular review cycle |

**Key ISO 42001 Controls for SOC AI**:

```python
ISO_42001_CONTROLS = {
    "A.2.2": "AI system impact assessment performed before deployment",
    "A.2.3": "Intended purpose of AI system documented and constrained",
    "A.3.3": "AI training data sources documented and validated",
    "A.5.1": "AI system transparency and explainability capabilities",
    "A.5.3": "Human oversight mechanisms implemented",
    "A.6.1": "AI system data quality management",
    "A.6.2": "Data minimization for AI training and inference",
    "A.7.4": "Robustness and security testing of AI systems",
    "A.8.3": "Feedback mechanisms for continuous AI improvement",
    "A.9.3": "Monitoring and logging of AI system operations"
}

def map_soc_controls_to_iso42001(soc_controls: list) -> dict:
    """Map SOC AI controls to ISO 42001 clauses"""
    mapping = {}
    for clause, description in ISO_42001_CONTROLS.items():
        matched_controls = [
            c for c in soc_controls
            if any(keyword in c['description'].lower() 
                   for keyword in description.lower().split())
        ]
        mapping[clause] = {
            "iso_requirement": description,
            "implemented_controls": matched_controls,
            "covered": len(matched_controls) > 0
        }
    return mapping
```
